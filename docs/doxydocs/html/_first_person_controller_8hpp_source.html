148 void updatePosition(float deltaT, const glm::vec3 &moveInput, std::vector<SceneObject> &scene) {
149 glm::vec3 walkDir = glm::normalize(glm::vec3(std::sin(yaw), 0.0F, -std::cos(yaw)));
150 glm::vec3 right = glm::normalize(glm::cross(walkDir, glm::vec3(0.0F, 1.0F, 0.0F)));
151 glm::vec3 desiredMove =
152 walkDir * MOVE_SPEED * deltaT * (-moveInput.z) + right * MOVE_SPEED * deltaT * moveInput.x;
153
154 glm::vec3 newPos = camPos;
155 glm::vec3 fullPos(camPos.x + desiredMove.x, camPos.y, camPos.z + desiredMove.z);
156
157 // Collision resolution with wall sliding. First try the full move; if it is
158 // blocked, retry each horizontal axis on its own. Walking diagonally into a
159 // wall then keeps the component parallel to the wall and drops only the one
160 // pushing into it — so the player slides along the surface instead of
161 // sticking. tryZ starts from newPos.x (the already-accepted X) so the two
162 // axes don't fight and we never tunnel through a corner.
163 if (!collides(fullPos, scene)) {
164 newPos = fullPos;
165 } else {
166 glm::vec3 tryX(camPos.x + desiredMove.x, camPos.y, camPos.z);
167 if (!collides(tryX, scene)) {
168 newPos.x = tryX.x;
169 }
170
171 glm::vec3 tryZ(newPos.x, camPos.y, camPos.z + desiredMove.z);
172 if (!collides(tryZ, scene)) {
173 newPos.z = tryZ.z;
174 }
175 }
176
177 camPos = newPos;
178 }

187 bool collides(glm::vec3 testPos, std::vector<SceneObject> &scene) const {
188 Collider playerCollider;
189 playerCollider.initAABB(-PLAYER_RADIUS, 0.0F, -PLAYER_RADIUS, PLAYER_RADIUS, EYE_HEIGHT,
190 PLAYER_RADIUS);
191 playerCollider.setWorldMatrix(
192 glm::translate(glm::mat4(1), glm::vec3(testPos.x, testPos.y - EYE_HEIGHT, testPos.z)));
193
194 for (auto &obj : scene) {
195 if (!obj.collidable) {
196 continue;
197 }
198 if (playerCollider.collidesWith(obj.collider)) {
199 return true;
200 }
201 }
202 return false;
