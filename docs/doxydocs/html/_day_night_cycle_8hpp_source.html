50 // One full day every 120 s. Continuous and fixed: the cycle is part of the
51 // ambience, not something the player drives.
52 static constexpr float SPEED = 1.0f / 120.0f;
53
54 // timeOfDay in [0,1): 0.0 = midnight, 0.25 = sunrise, 0.5 = noon, 0.75 = sunset.
55 // Starts mid-morning so the scene opens comfortably lit.
56 float timeOfDay = 0.30f;
57
59 State sample() const {
60 // The sun sweeps a half-circle above the horizon between sunrise and sunset.
61 // phase = 0 at sunrise, pi/2 at noon, pi at sunset; elevation = sin(phase)
62 // so it is 0 on the horizon, 1 overhead, and negative through the night.
63 const float phase = (timeOfDay - 0.25f) * glm::two_pi<float>(); //Xonverts [0,1) to (-pi/2 , 3pi/2)
64 const float elevation = std::sin(phase); //Sun height
65
66 // East (-X) at dawn, overhead at noon, west (+X) at dusk, with a slight
67 // constant tilt on Z so the arc is not a dead-vertical plane.
68 glm::vec3 toSun = glm::normalize(glm::vec3(-std::cos(phase), elevation, 0.20f));
69
70 // Fade the directional light in as the sun clears the horizon and out as it
71 // sets; dayMix additionally separates low warm sun from high white daylight.
72 // The fade starts exactly at the horizon (elevation 0), never below it: a sun
73 // with elevation < 0 would shine *upward* through the ground plane, unshadowed
74 // (its shadow strength is also 0 there), producing a flash as it crosses the
75 // horizon at sunrise/sunset. Keeping it above 0 ties intensity and shadow
76 // together so the transition is smooth.
77 const float aboveFade = smoothstep(0.0f, 0.15f, elevation);
78 // Slow ramp so the yellow-orange tint dominates most of the arc, only giving
79 // way to white-yellow once the sun is high overhead.
80 const float dayMix = smoothstep(0.0f, 0.75f, elevation);
81
82 const glm::vec3 warmSun = glm::vec3(1.00f, 0.45f, 0.05f); // low sun: deep amber-orange
83 const glm::vec3 noonSun = glm::vec3(1.00f, 0.95f, 0.72f); // zenith: warm white-yellow
84 const glm::vec3 moon = glm::vec3(0.45f, 0.55f, 0.85f); // night: cool blue-white
85
86 State s{};
87 s.toSun = toSun;
88 s.sunDir = -toSun;
89 s.color = glm::mix(moon, glm::mix(warmSun, noonSun, dayMix), aboveFade);
90 // No exterior directional light at night: the moon casts nothing, so the only
91 // illumination outdoors comes from the tavern's own torches spilling through
92 // its windows and doorway (their point lights already reach outside). Daytime
93 // ramps from a warm low sun to the midday sun. A directional light has no
94 // distance attenuation, so it must stay on the same scale as the torches'
95 // point lights (intensity ~0.6-1.0) or it floods the whole interior.
96 s.intensity = glm::mix(0.0f, glm::mix(0.45f, 1.2f, dayMix), aboveFade);
97 s.dayFactor = aboveFade;
98 return s;
99 }
100
107 static float smoothstep(float edge0, float edge1, float x) {
108 float t = glm::clamp((x - edge0) / (edge1 - edge0), 0.0f, 1.0f);
109 return t * t * (3.0f - 2.0f * t);
110 }
