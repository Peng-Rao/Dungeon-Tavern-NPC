33 void load(const std::string &path) {
34 std::ifstream file(path);
35 if (!file.is_open()) {
36 return;
37 }
38 nlohmann::json dialogueJson;
39 file >> dialogueJson;
40 for (const auto &[npcId, treeJson] : dialogueJson.items()) {
41 Tree tree;
42 tree.name = treeJson.value("name", npcId);
43 tree.start = treeJson.value("start", "root");
44 for (const auto &[nodeId, nodeJson] : treeJson["nodes"].items()) {
45 Node node;
46 node.text = nodeJson.value("text", "");
47 for (const auto &choiceJson : nodeJson.value("choices", nlohmann::json::array())) {
48 node.choices.push_back({choiceJson.value("label", ""), choiceJson.value("next", ""),
49 choiceJson.value("action", "")});
50 }
51 tree.nodes[nodeId] = std::move(node);
52 }
53 trees[npcId] = std::move(tree);
54 }

64 void update(const std::string &npcId, bool interactPressed,
65 const std::array<bool, 3> &choicePressed) {
66 if (interactPressed) {
67 if (open) {
68 close();
69 return;
70 }
71 if (!npcId.empty() && trees.count(npcId) != 0) {
72 open = true;
73 activeNpc = npcId;
74 nodeId = trees[npcId].start;
75 }
76 }
77 if (!open) {
78 return;
79 }
80 // Walking or looking away ends the conversation.
81 if (npcId != activeNpc) {
82 close();
83 return;
84 }
85 const Node *node = currentNode();
86 if (node == nullptr) {
87 close();
88 return;
89 }
90 for (int i = 0; i < (int)node->choices.size() && i < 3; i++) {
91 if (!choicePressed[i]) {
92 continue;
93 }
94 const Choice &choice = node->choices[i];
95 if (choice.action == "shop") {
96 shopRequest = true;
97 close();
98 } else if (choice.next.empty()) {
99 close();
100 } else {
101 nodeId = choice.next;
102 }
103 break;
104 }
105 }
