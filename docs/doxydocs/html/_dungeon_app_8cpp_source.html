402 glm::mat4 computeSpotLightVP(const Light &L) const {
403 glm::vec3 p = glm::vec3(L.pos);
404 glm::vec3 axis = glm::normalize(glm::vec3(L.dir));
405 float outerCos = glm::clamp(L.cones.y, -1.0f, 1.0f);
406 float fov = glm::min(2.0f * std::acos(outerCos) + glm::radians(8.0f), glm::radians(160.0f));
407 float farP = (L.color.a > 0.0f) ? L.color.a : 10.0f;
408 glm::vec3 up = (std::abs(axis.y) > 0.99f) ? glm::vec3(0, 0, 1) : glm::vec3(0, 1, 0);
409 glm::mat4 view = glm::lookAt(p, p + axis, up);
410 glm::mat4 proj = glm::perspective(fov, 1.0f, 0.05f, farP);
411 // Vulkan clip-space Y flip, applied as a scale on the projection (the course
412 // convention), exactly like the sun's matrix and the main camera.
413 proj = glm::scale(glm::mat4(1.0f), glm::vec3(1.0f, -1.0f, 1.0f)) * proj;
414 return proj * view;
415 }

454 std::string texturePathForModel(const std::string &modelPath) const {
455 std::ifstream file(modelPath);
456 if (!file.is_open()) return "";
457 nlohmann::json gltf;
458 try {
459 file >> gltf;
460 } catch (const std::exception &) {
461 return "";
462 }
463 if (!gltf.contains("images") || gltf["images"].empty()) return "";
464 const auto &image = gltf["images"][0];
465 if (!image.contains("uri")) return "";
466 const std::string uri = image["uri"].get<std::string>();
467 if (uri.empty() || uri.rfind("data:", 0) == 0) return ""; // skip embedded data URIs
468 return (std::filesystem::path(modelPath).parent_path() / uri).string();
469 }

685 void localInit() {
686 // PERF: the framework picks the GPU's MAXIMUM MSAA level (often 8x) and the
687 // pipeline forces per-sample shading, so the fragment shader runs once per
688 // sample at 8x that is 8x the per-pixel cost, which on its own can drag the
689 // framerate down regardless of how light the geometry is. msaaSamples is a
690 // public framework field, so we can cap it here (before the render pass and
691 // pipelines are built) without editing the framework itself. 4x still looks
692 // clean; drop to 2x (or VK_SAMPLE_COUNT_1_BIT) if more speed is needed.
693 if (msaaSamples > VK_SAMPLE_COUNT_4_BIT) {
694 msaaSamples = VK_SAMPLE_COUNT_4_BIT;
695 }
696
697 DSLlocalTextured.init(this,
698 {{0, VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VK_SHADER_STAGE_ALL_GRAPHICS,
699 sizeof(UniformBufferObject), 1},
700 {1, VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER,
701 VK_SHADER_STAGE_FRAGMENT_BIT, 0, 1}});
702
703 DSLglobal.init(this, {{0, VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VK_SHADER_STAGE_ALL_GRAPHICS,
704 sizeof(GlobalUniformBufferObject), 1},
705 // binding 1: the sun's 2D shadow map (E07-style directional
706 // depth map). The 4th field is this binding's start offset
707 // into the image-info vector passed to DescriptorSet::init.
708 {1, VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, VK_SHADER_STAGE_FRAGMENT_BIT,
709 0, 1},
710 // binding 2: the per-spotlight 2D shadow maps, packed right
711 // after the sun map (offset 1) in the image-info vector.
712 {2, VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, VK_SHADER_STAGE_FRAGMENT_BIT,
713 1, MAX_SHADOW_SPOTS}});
714
715 // Skybox set: binding 0 the per-frame UBO (mvp + sun params, read in both
716 // stages), binding 1 the sky cube map.
717 DSLskybox.init(this, {{0, VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, VK_SHADER_STAGE_ALL_GRAPHICS,
718 sizeof(SkyboxUBO), 1},
719 {1, VK_DESCRIPTOR_TYPE_COMBINED_IMAGE_SAMPLER, VK_SHADER_STAGE_FRAGMENT_BIT,
720 0, 1}});
721
722 Tdungeon.init(this, "assets/textures/dungeon/dungeon_texture.png");
723 Tground.init(this, "assets/textures/ground/ground.png");
724 // Sky cube map (6 faces). Order expected by initCubic: +X,-X,+Y,-Y,+Z,-Z.
725 TenvMap.initCubic(this, {"assets/textures/skybox/px.png", "assets/textures/skybox/nx.png",
726 "assets/textures/skybox/py.png", "assets/textures/skybox/ny.png",
727 "assets/textures/skybox/pz.png", "assets/textures/skybox/nz.png"});
728 setApplicationIcon();
729
730 VDsimple.init(
731 this, {{0, sizeof(VertexSimple), VK_VERTEX_INPUT_RATE_VERTEX}},
732 {{0, 0, VK_FORMAT_R32G32B32_SFLOAT, offsetof(VertexSimple, pos), sizeof(glm::vec3), POSITION},
733 {0, 1, VK_FORMAT_R32G32B32_SFLOAT, offsetof(VertexSimple, norm), sizeof(glm::vec3), NORMAL},
734 {0, 2, VK_FORMAT_R32G32_SFLOAT, offsetof(VertexSimple, UV), sizeof(glm::vec2), UV}});
735
736 // The skybox reads only positions, but from the same cube buffer (stride is
737 // still a full VertexSimple). A second, leaner vertex format on the same
738 // mesh — exactly the "different formats for different needs" idea.
739 VDskybox.init(this, {{0, sizeof(VertexSimple), VK_VERTEX_INPUT_RATE_VERTEX}},
740 {{0, 0, VK_FORMAT_R32G32B32_SFLOAT, offsetof(VertexSimple, pos),
741 sizeof(glm::vec3), POSITION}});
742
743 RP.init(this);
744 RP.properties[0].clearValue = {0.01f, 0.01f, 0.02f, 1.0f};
745
746 Psimple.init(this, &VDsimple, "shaders/mesh/MeshSimple.vert.spv",
747 "shaders/mesh/BlinnPhong.frag.spv", {&DSLglobal, &DSLlocalTextured});
748
749 // Skybox pipeline: we are inside the cube (cull front), and depth-test with
750 // LESS_OR_EQUAL so the sky (pushed to the far plane) survives only where no
751 // opaque geometry already wrote depth.
752 Pskybox.init(this, &VDskybox, "shaders/skybox/Skybox.vert.spv",
753 "shaders/skybox/Skybox.frag.spv", {&DSLskybox});
754 Pskybox.setCompareOp(VK_COMPARE_OP_LESS_OR_EQUAL);
755 Pskybox.setCullMode(VK_CULL_MODE_FRONT_BIT);
756
757 // Offscreen depth-only pass for the sun's shadow map. Fixed size (it covers
758 // the whole scene, not the window), so it is independent of the swap chain.
759 RPsun.init(this, SUN_SHADOW_RES, SUN_SHADOW_RES, 1,
760 RenderPass::getStandardAttchmentsProperties(AT_DEPTH_ONLY, this),
761 RenderPass::getStandardDependencies(ATDEP_DEPTH_TRANS), true);
762
763 // The depth pass only needs each object's model matrix (set 0 = its UBO) and
764 // the sun's view-projection (a push constant). Cull NONE because dungeon
765 // walls can be single-sided planes — culling could drop them from the depth
766 // map and leak light through them.
767 // A depth-only pass needs only the vertex position, so it uses the

1057 void updateUniformBuffer(uint32_t currentImage) {
1058 float deltaT = GameLogic();
1059
1060 animTime += deltaT;
1061 sunState = dayNight.update(deltaT); // advance the continuous day/night cycle
1062
1063 // Swing doors toward their target pose with an exponential ease (fast
1064 // start, gentle settle), snapping once the leaf is within half a degree.
1065 // A moving leaf does not collide so a swinging door can never trap the
1066 // player mid-sweep the collider comes back once it rests.
1067 for (auto &obj : scene) {
1068 if (!obj.isDoor) continue;
1069 float target = obj.doorOpen ? obj.openYaw : obj.closedYaw;
1070 float diff = target - obj.yaw;
1071 if (std::abs(diff) < 0.5f) {
1072 if (!obj.collidable) {
1073 obj.yaw = target;
1074 obj.collidable = true;
1075 refreshColliderWorld(obj);
1076 }
1077 continue;
1078 }
1079 obj.yaw += diff * std::min(1.0f, DOOR_SWING_RATE * deltaT);
1080 obj.collidable = false;
1081 }
1082
1083 // NPCs turn to face the player while spoken to (dialogue or shop) or while
1084 // the player stands in their personal space. When not engaged, patrolling
1085 // NPCs walk their waypoint loop (facing the walk direction) and stationary
1086 // NPCs ease back to their authored resting facing.
1087 for (auto &obj : scene) {
1088 const bool isNpc = !obj.npcId.empty();
1089 if (!isNpc && obj.patrolPoints.empty()) continue;
1090 glm::vec3 toPlayer = cameraPos - obj.pos;
1091 toPlayer.y = 0.0f;
1092 float playerDist = glm::length(toPlayer);
1093 const bool talking =
1094 (dialogueSystem.isOpen() && dialogueSystem.activeNpcId() == obj.npcId) ||
1095 (shopSystem.isOpen() && obj.npcId == shopNpcId);
1096 if (talking || playerDist < NPC_PERSONAL_SPACE) {
1097 if (playerDist > 0.01f) {
1098 turnNpcToward(obj, glm::degrees(std::atan2(toPlayer.x, toPlayer.z)), deltaT);
1099 }
1100 continue;
1101 }
1102 // Not engaged: a stationary NPC returns to its authored facing.
1103 if (obj.patrolPoints.empty()) {
1104 turnNpcToward(obj, obj.restYaw, deltaT);
1105 continue;
1106 }
1107 if (obj.patrolWait > 0.0f) {
1108 obj.patrolWait -= deltaT;
1109 continue;
1110 }
1111 glm::vec3 toTarget = obj.patrolPoints[obj.patrolTarget] - obj.pos;
1112 toTarget.y = 0.0f;
1113 float dist = glm::length(toTarget);
1114 if (dist < 0.05f) {
1115 obj.patrolTarget = (obj.patrolTarget + 1) % (int)obj.patrolPoints.size();
1116 obj.patrolWait = obj.patrolPause;
1117 continue;
1118 }
1119 glm::vec3 dir = toTarget / dist;
1120 glm::vec3 step = dir * std::min(dist, obj.patrolSpeed * deltaT);
1121 obj.pos += step;
1122 obj.boundsCenter += step; // keep the shadow-cull sphere with the walker
1123 turnNpcToward(obj, glm::degrees(std::atan2(dir.x, dir.z)), deltaT);
1124 }
1125
1126 // A carried torch rides with the camera; refresh its pose before the light
1127 // list and the per-object matrices below are built from it.
1128 updateHeldTorch();
1129
1130 GlobalUniformBufferObject gubo{};
1131
1132 // Rebuild the GPU light list from scratch every frame: each lit flame adds
1133 // its light. Lighting or snuffing a candle just flips its `lit` flag, so
1134 // there are no stale slots to manage. Spotlight shadow slots are reassigned
1135 // here too, so lighting a torch makes it start casting and snuffing it frees
1136 // the slot for another.
1137 int activeLights = 0;
1138 activeSpotShadows = 0;
1139 for (int oi = 0; oi < (int)scene.size(); oi++) {
1140 auto &obj = scene[oi];
1141 if (!obj.isFlame) continue;
1142
1143 if (!obj.lit) {
1144 obj.emissive = glm::vec3(0.0f); // snuffed: no glow, contributes no light
1145 continue;
1146 }
1147
1148 // ---- Flame flicker ----
1149 // Wobble around the resting brightness with two sine waves at unrelated
1150 // frequencies (11 and 6.3): a single sine looks like a steady mechanical
1151 // pulse, but two detuned ones never line up the same way twice, so the eye
1152 // reads it as the random dance of a real flame. We only ever *dim* (factor
1153 // in [0.8, 1.0]) so the flame never flashes brighter than its design value.
1154 float wobble = 0.5f * std::sin(animTime * 11.0f + obj.flamePhase) +
1155 0.5f * std::sin(animTime * 6.3f + obj.flamePhase * 2.1f);
1156 float factor = 0.9f + 0.1f * wobble; // [-1,1] -> [0.8,1.0]
1157
1158 obj.emissive = obj.baseEmissive * factor;
1159
1160 if (activeLights < MAX_LIGHTS) {
1161 Light L = obj.light;
1162 L.dir.w = obj.baseIntensity * factor; // flickered intensity
1163
1164 // A lit torch (spotlight) gets a 2D shadow map while a free slot remains.
1165 // cones.z carries that slot index to the fragment shader; -1 = this
1166 // spotlight casts no shadow this frame (slots exhausted).
1167 if (static_cast<int>(L.pos.w) == LIGHT_SPOT) {
1168 L.cones.z = -1.0f;
1169 if (activeSpotShadows < MAX_SHADOW_SPOTS) {
1170 int slot = activeSpotShadows++;
1171 glm::mat4 vp = computeSpotLightVP(L);
1172 spotShadows[slot].lightVP = vp;
1173 spotShadows[slot].emitterIndex = oi;
1174 gubo.spotLightVP[slot] = vp;
1175 L.cones.z = static_cast<float>(slot);
1176 }
1177 }
1178
1179 gubo.lights[activeLights++] = L;
1180
1181 // A spotlight alone leaves the torch's own wall and immediate surroundings
1182 // dark, but a real flame glows in every direction. So each torch also emits
1183 // a dim, short-range omnidirectional POINT "fill" co-located with the flame:
1184 // it lights the wall and nearby props (no cone, no shadow) while the
1185 // spotlight above does the directional pool that actually casts shadows.
1186 if (static_cast<int>(L.pos.w) == LIGHT_SPOT && activeLights < MAX_LIGHTS) {
1187 Light fill{};
1188 fill.pos = glm::vec4(glm::vec3(L.pos), static_cast<float>(LIGHT_POINT));
1189 fill.dir = glm::vec4(0.0f, 0.0f, 0.0f, L.dir.w * 0.35f); // dimmer than the spot
1190 fill.color = glm::vec4(glm::vec3(L.color), 1.8f); // warm tint, ~1.8 m reach
1191 fill.cones = glm::vec4(0.0f); // point light: no cone/shadow
1192 gubo.lights[activeLights++] = fill;
1193 }
1194 }
1195 }
1196
1197 // Orthographic view-projection from the sun, framed on the whole scene, used
1198 // both to render the depth map and to look it up in the main shader. The Y
1199 // axis is flipped in the ortho (top/bottom swapped) so the sampled UVs line
1200 // up with how the depth map was rasterised. Computed every frame so the
1201 // depth pass is always valid, even when the result is not applied at night.
1202 {
1203 glm::vec3 sDir = glm::normalize(sunState.sunDir);
1204 glm::vec3 up = (std::abs(sDir.y) > 0.99f) ? glm::vec3(0, 0, 1) : glm::vec3(0, 1, 0);
1205 float dist = 2.0f * sceneRadius;
1206 glm::mat4 lightView = glm::lookAt(sceneCenter - sDir * dist, sceneCenter, up);
1207 glm::mat4 lightProj = glm::ortho(-sceneRadius, sceneRadius, -sceneRadius, sceneRadius,
1208 0.0f, 2.0f * dist);
1209 // Vulkan's clip-space Y is flipped relative to OpenGL. As with the main
1210 // camera, apply the flip as a scale on the projection (the course's
1211 // convention) rather than swapping the ortho's top/bottom.
1212 lightProj = glm::scale(glm::mat4(1.0f), glm::vec3(1.0f, -1.0f, 1.0f)) * lightProj;
1213 sunLightVP = lightProj * lightView;
1214 }
1215 gubo.sunLightVP = sunLightVP;
1216
1217 // The sun/moon: a single directional light driven by the day/night cycle.
1218 // Direction, colour and intensity all come from the cycle, so the scene
1219 // brightens at dawn, warms at dusk and dims to cool moonlight at night.
1220 // range (color.a) = 0 means infinite (a directional light never attenuates).
1221 // cones.z >= 0 tells the shader to clip this light with the sun shadow map;
1222 // we only enable it while the sun is above the horizon (its shadow would
1223 // come from below at night), but it stays a lit (dim) moon either way.
1224 if (activeLights < MAX_LIGHTS && sunState.intensity > 0.001f) {
1225 // Continuous shadow strength (0..1) instead of an on/off flag: it ramps up
1226 // as the sun clears the horizon and back down as it sets, so the shadowing
1227 // never switches abruptly. Paired with the intensity fade, this removes the
1228 // bright "flash" that the old boolean toggle produced at sunset.
1229 const float shadowStrength = glm::clamp(sunState.toSun.y / 0.12f, 0.0f, 1.0f);
1230 Light sun{};
1231 sun.pos = glm::vec4(0.0f, 0.0f, 0.0f, (float)LIGHT_DIRECTIONAL);
1232 sun.dir = glm::vec4(glm::normalize(sunState.sunDir), sunState.intensity);
1233 sun.color = glm::vec4(sunState.color, 0.0f);
1234 sun.cones = glm::vec4(0.0f, 0.0f, shadowStrength, 0.0f);
1235 gubo.lights[activeLights++] = sun;
1236 }
1237
1238 gubo.eyePos = glm::vec4(cameraPos, (float)activeLights);
1239
1240 DSglobal.map(currentImage, &gubo, 0);
1241
1242 // Skybox: rotation-only view-projection (so it stays centred on the camera),
1243 // plus the sun direction/day factor/colour that tint it across the cycle.
1244 SkyboxUBO skyUbo{};
1245 skyUbo.mvpMat = SkyViewPrj;
1246 skyUbo.sunDirDay = glm::vec4(sunState.toSun, sunState.dayFactor);
1247 skyUbo.sunColor = glm::vec4(sunState.color, 1.0f);
1248 DSskybox.map(currentImage, &skyUbo, 0);
1249
1250 for (auto &obj : scene) {
1251 UniformBufferObject ubo{};
1252 ubo.mMat = glm::translate(glm::mat4(1), obj.pos) *
1253 glm::rotate(glm::mat4(1), glm::radians(obj.yaw), glm::vec3(0, 1, 0)) *
1254 glm::scale(glm::mat4(1), glm::vec3(obj.scale)) *
1255 obj.model->Wm;
1256 ubo.mvpMat = ViewPrj * ubo.mMat;
1257 ubo.nMat = glm::inverse(glm::transpose(ubo.mMat));
1258 ubo.matParams = glm::vec4(obj.specExp, obj.emissive.r, obj.emissive.g, obj.emissive.b);
1259 obj.DS.map(currentImage, &ubo, 0);
1260 }
1261
1262 static float elapsedT = 0.0f;
1263 static int countedFrames = 0;
1264 countedFrames++;
1265 elapsedT += deltaT;
1266 if (elapsedT > 1.0f) {
1267 lastFps = (float)countedFrames / elapsedT;
1268 elapsedT = 0.0f;
1269 countedFrames = 0;
1270 }
1271
1272 lastDeltaTime = deltaT;
1273 buildImGuiFrame();
1274 submitCommandBuffer("main", 0, populateCommandBufferAccess, this);
1275 }

1286 float GameLogic() {
1287 const float FOVy = glm::radians(60.0f);
1288 const float nearPlane = 0.1f;
1289 const float farPlane = 100.0f;
1290 const float ORBIT_MOUSE_SENS = 0.005f; // overhead camera: radians per pixel
1291 const float ZOOM_RATE = 0.8f; // overhead camera: frame change per second
1292 const float ZOOM_MIN = 0.25f; // closest zoom (tightest ortho frame)
1293 const float ZOOM_MAX = 2.0f; // farthest zoom (widest ortho frame)
1294 const float INTERACT_DIST = 2.5f;
1295 const float INTERACT_DOT = 0.6f;
1296
1297 float deltaT;
1298 glm::vec3 m(0.0f), r(0.0f);
1299 bool fire = false;
1300 getSixAxis(deltaT, m, r, fire); //Gets the motion axes in m and the rotation axes in r and how pressed was with fire
1301
1302 // While the splash menu is up it owns all input: the player stands still,

1498 // Ar) so the world is not stretched on wide or tall windows.

1500 const float halfW = halfH * Ar;

1503 // orbit distance only sets the view angle and clipping (ortho size is
1504 // distance-independent).
1505 const float dist = sceneRadius * 2.5f;
1506 const glm::vec3 eye =

1509 std::cos(overheadPitch) * std::sin(overheadYaw));

1511 } else {

1513 View = firstPersonController.viewMatrix();
1514 }

1516 ViewPrj = Prj * View;
1517 SkyViewPrj = Prj * glm::mat4(glm::mat3(View)); // drop translation: sky tracks the camera
1518
1519 return deltaT;
